116void DGRPacketSink::StartApplication () // Called at time specified by Start
117{
118 NS_LOG_FUNCTION (this);
119 // Create the socket if not already
120 if (!m_socket)
121 {
122 m_socket = Socket::CreateSocket (GetNode (), m_tid);
123 if (m_socket->Bind (m_local) == -1)
124 {
125 NS_FATAL_ERROR ("Failed to bind socket");
126 }
127 m_socket->Listen ();
128 m_socket->ShutdownSend ();
129 if (addressUtils::IsMulticast (m_local))
130 {
131 Ptr<UdpSocket> udpSocket = DynamicCast<UdpSocket> (m_socket);
132 if (udpSocket)
133 {
134 // equivalent to setsockopt (MCAST_JOIN_GROUP)
135 udpSocket->MulticastJoinGroup (0, m_local);
136 }
137 else
138 {
139 NS_FATAL_ERROR ("Error: joining multicast on a non-UDP socket");
140 }
141 }
142 }
143
144 m_socket->SetRecvCallback (MakeCallback (&DGRPacketSink::HandleRead, this));
145 m_socket->SetAcceptCallback (
146 MakeNullCallback<bool, Ptr<Socket>, const Address &> (),
147 MakeCallback (&DGRPacketSink::HandleAccept, this));
148 m_socket->SetCloseCallbacks (
149 MakeCallback (&DGRPacketSink::HandlePeerClose, this),
150 MakeCallback (&DGRPacketSink::HandlePeerError, this));
151}

243void
244DGRPacketSink::PacketReceived (const Ptr<Packet> &p, const Address &from,
245 const Address &localAddress)
246{
247 SeqTsSizeHeader header;
248 Ptr<Packet> buffer;
249
250 auto itBuffer = m_buffer.find (from);
251 if (itBuffer == m_buffer.end ())
252 {
253 itBuffer = m_buffer.insert (std::make_pair (from, Create<Packet> (0))).first;
254 }
255
256 buffer = itBuffer->second;
257 buffer->AddAtEnd (p);
258 buffer->PeekHeader (header);
259
260 NS_ABORT_IF (header.GetSize () == 0);
261
262 while (buffer->GetSize () >= header.GetSize ())
263 {
264 NS_LOG_DEBUG ("Removing packet of size " << header.GetSize () << " from buffer of size " << buffer->GetSize ());
265 Ptr<Packet> complete = buffer->CreateFragment (0, static_cast<uint32_t> (header.GetSize ()));
266 buffer->RemoveAtStart (static_cast<uint32_t> (header.GetSize ()));
267
268 complete->RemoveHeader (header);
269
270 m_rxTraceWithSeqTsSize (complete, from, localAddress, header);
271
272 if (buffer->GetSize () > header.GetSerializedSize ())
273 {
274 buffer->PeekHeader (header);
275 }
276 else
277 {
278 break;
279 }
280 }
