271std::ostream& operator<< (std::ostream& os, Ipv4DGRRoutingTableEntry const& route)
272{
273 if (route.IsDefault ())
274 {
275 NS_ASSERT (route.IsGateway ());
276 os << "default out=" << route.GetInterface () << ", next hop=" << route.GetGateway ();
277 }
278 else if (route.IsHost ())
279 {
280 if (route.IsGateway ())
281 {
282 os << "host="<< route.GetDest () <<
283 ", out=" << route.GetInterface () <<
284 ", next hop=" << route.GetGateway ();
285 }
286 else
287 {
288 os << "host="<< route.GetDest () <<
289 ", out=" << route.GetInterface ();
290 }
291 }
292 else if (route.IsNetwork ())
293 {
294 if (route.IsGateway ())
295 {
296 os << "network=" << route.GetDestNetwork () <<
297 ", mask=" << route.GetDestNetworkMask () <<
298 ",out=" << route.GetInterface () <<
299 ", next hop=" << route.GetGateway ();
300 }
301 else
302 {
303 os << "network=" << route.GetDestNetwork () <<
304 ", mask=" << route.GetDestNetworkMask () <<
305 ",out=" << route.GetInterface ();
306 }
307 }
308 else
309 {
310 NS_ASSERT (false);
311 }
312 return os;
313}
