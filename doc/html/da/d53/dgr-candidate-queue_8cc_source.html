172/*
173 * In this implementation, DGRVertex follows the ordering where
174 * a vertex is ranked first if its GetDistanceFromRoot () is smaller;
175 * In case of a tie, NetworkLSA is always ranked before RouterLSA.
176 *
177 * This ordering is necessary for implementing ECMP
178 */
179bool
180DGRCandidateQueue::CompareDGRVertex (const DGRVertex* v1, const DGRVertex* v2)
181{
182 NS_LOG_FUNCTION (&v1 << &v2);
183
184 bool result = false;
185 if (v1->GetDistanceFromRoot () < v2->GetDistanceFromRoot ())
186 {
187 result = true;
188 }
189 else if (v1->GetDistanceFromRoot () == v2->GetDistanceFromRoot ())
190 {
191 if (v1->GetVertexType () == DGRVertex::VertexNetwork
192 && v2->GetVertexType () == DGRVertex::VertexRouter)
193 {
194 result = true;
195 }
196 }
197 return result;
198}
