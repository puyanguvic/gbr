144void
145DgrHeader::Serialize (Buffer::Iterator start) const
146{
147 Buffer::Iterator i = start;
148 i.WriteU8 (uint8_t(m_command));
149 i.WriteU8 (2);
150 i.WriteU16 (0);
151
152 for (std::list<DgrNse>::const_iterator iter = m_nseList.begin ();
153 iter != m_nseList.end ();
154 iter ++)
155 {
156 iter->Serialize (i);
157 i.Next (iter->GetSerializedSize ());
158 }

161uint32_t
162DgrHeader::Deserialize (Buffer::Iterator start)
163{
164 Buffer::Iterator i = start;
165
166 uint8_t temp;
167 temp = i.ReadU8 ();
168 if ((temp == REQUEST) || (temp == RESPONSE))
169 {
170 m_command = temp;
171 }
172 else
173 {
174 return 0;
175 }
176
177 if (i.ReadU8 () != 2)
178 {
179 NS_LOG_LOGIC ("DGR received a message with mismatch version, ignoring.");
180 return 0;
181 }
182
183 if (i.ReadU16 () != 0)
184 {
185 NS_LOG_LOGIC ("DGR received a message with invalid filled flags, ignoring.");
186 return 0;
187 }
188
189 DgrNse nse;
190 uint32_t nseSize = nse.GetSerializedSize ();
191 uint8_t nseNumber = i.GetRemainingSize ()/ nseSize; // !!!!!!!!!!!!! the size should be the same with nse.
192 for (uint8_t n = 0; n < nseNumber; n ++)
193 {
194 i.Next (nse.Deserialize (i));
195 m_nseList.push_back (nse);
196 }
197
198 return GetSerializedSize ();
199}
