143void DGRTcpApplication::StartApplication (void) // Called at time specified by Start
144{
145 NS_LOG_FUNCTION (this);
146 Address from;
147 // Create the socket if not already
148 if (!m_socket)
149 {
150 m_socket = Socket::CreateSocket (GetNode (), m_tid);
151 int ret = -1;
152
153 // Fatal error if socket type is not NS3_SOCK_STREAM or NS3_SOCK_SEQPACKET
154 if (m_socket->GetSocketType () != Socket::NS3_SOCK_STREAM &&
155 m_socket->GetSocketType () != Socket::NS3_SOCK_SEQPACKET)
156 {
157 NS_FATAL_ERROR ("Using BulkSend with an incompatible socket type. "
158 "BulkSend requires SOCK_STREAM or SOCK_SEQPACKET. "
159 "In other words, use TCP instead of UDP.");
160 }
161
162 if (! m_local.IsInvalid())
163 {
164 NS_ABORT_MSG_IF ((Inet6SocketAddress::IsMatchingType (m_peer) && InetSocketAddress::IsMatchingType (m_local)) ||
165 (InetSocketAddress::IsMatchingType (m_peer) && Inet6SocketAddress::IsMatchingType (m_local)),
166 "Incompatible peer and local address IP version");
167 ret = m_socket->Bind (m_local);
168 }
169 else
170 {
171 if (Inet6SocketAddress::IsMatchingType (m_peer))
172 {
173 ret = m_socket->Bind6 ();
174 }
175 else if (InetSocketAddress::IsMatchingType (m_peer))
176 {
177 ret = m_socket->Bind ();
178 }
179 }
180
181 if (ret == -1)
182 {
183 NS_FATAL_ERROR ("Failed to bind socket");
184 }
185
186 m_socket->Connect (m_peer);
187 m_socket->ShutdownRecv ();
188 m_socket->SetConnectCallback (
189 MakeCallback (&DGRTcpApplication::ConnectionSucceeded, this),
190 MakeCallback (&DGRTcpApplication::ConnectionFailed, this));
191 m_socket->SetSendCallback (
192 MakeCallback (&DGRTcpApplication::DataSend, this));
193 }
194 if (m_connected)
195 {
196 m_socket->GetSockName (from);
197 SendData (from, m_peer);
198 }
199}

219void DGRTcpApplication::SendData (const Address &from, const Address &to)
220{
221 NS_LOG_FUNCTION (this);
222
223 while (m_maxBytes == 0 || m_totBytes < m_maxBytes)
224 { // Time to send more
225
226 // uint64_t to allow the comparison later.
227 // the result is in a uint32_t range anyway, because
228 // m_sendSize is uint32_t.
229 uint64_t toSend = m_sendSize;
230 // Make sure we don't send too many
231 if (m_maxBytes > 0)
232 {
233 toSend = std::min (toSend, m_maxBytes - m_totBytes);
234 }
235
236 NS_LOG_LOGIC ("sending packet at " << Simulator::Now ());
237 Ptr<Packet> packet;
238
239 TimestampTag txTimeTag;
240 FlagTag flagTag;
241 BudgetTag budgetTag;
242 txTimeTag.SetTimestamp (Simulator::Now ());
243 flagTag.SetFlag (m_flag);
244 // flagTag.Print (std::cout);
245
246 if (m_unsentPacket)
247 {
248 packet = m_unsentPacket;
249 toSend = packet->GetSize ();
250 }
251 else
252 {
253 packet = Create<Packet> (toSend);
254 packet->AddPacketTag (txTimeTag);
255 packet->AddPacketTag (flagTag);
256 if (m_budget != MAX_UINT_32)
257 {
258 budgetTag.SetBudget (m_budget);
259 packet->AddPacketTag (budgetTag);
260 }
261 }
262 int actual = m_socket->Send (packet);
263 if ((unsigned) actual == toSend)
264 {
265 m_totBytes += actual;
266 m_txTrace (packet);
267 m_unsentPacket = 0;
268 }
269 else if (actual == -1)
270 {
271 // We exit this loop when actual < toSend as the send side
272 // buffer is full. The "DataSent" callback will pop when
273 // some buffer space has freed up.
274 NS_LOG_DEBUG ("Unable to send packet; caching for later attempt");
275 m_unsentPacket = packet;
276 break;
277 }
278 else if (actual > 0 && (unsigned) actual < toSend)
279 {
280 // A Linux socket (non-blocking, such as in DCE) may return
281 // a quantity less than the packet size. Split the packet
282 // into two, trace the sent packet, save the unsent packet
283 NS_LOG_DEBUG ("Packet size: " << packet->GetSize () << "; sent: " << actual << "; fragment saved: " << toSend - (unsigned) actual);
284 Ptr<Packet> sent = packet->CreateFragment (0, actual);
285 Ptr<Packet> unsent = packet->CreateFragment (actual, (toSend - (unsigned) actual));
286 m_totBytes += actual;
287 m_txTrace (sent);
288 m_unsentPacket = unsent;
289 break;
290 }
291 else
292 {
293 NS_FATAL_ERROR ("Unexpected return value from m_socket->Send ()");
294 }
295 }
296 // Check if time to close (all sent)
297 if (m_totBytes == m_maxBytes && m_connected)
298 {
299 m_socket->Close ();
300 m_connected = false;
301 }
302}
