120void
121DGRUdpApplication::SendPacket()
122{
123 TimestampTag txTimeTag;
124 FlagTag flagTag;
125 BudgetTag budgetTag;
126 PriorityTag priorityTag;
127
128 Ptr<Packet> packet = Create <Packet> (m_packetSize);
129 Time txTime = Simulator::Now ();
130 if (m_budget != MAX_UINT_32)
131 {
132 budgetTag.SetBudget (m_budget);
133 priorityTag.SetPriority (1);
134 packet->AddPacketTag (budgetTag);
135 packet->AddPacketTag (priorityTag);
136 }
137 flagTag.SetFlag (m_flag);
138 txTimeTag.SetTimestamp (txTime);
139 packet->AddPacketTag (txTimeTag);
140 packet->AddPacketTag (flagTag);
141 m_socket->Send (packet);
142 if(++ m_packetSent < m_nPackets)
143 {
144 ScheduleTx ();
145 }
146}
