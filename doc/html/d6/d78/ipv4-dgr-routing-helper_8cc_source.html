45Ptr<Ipv4RoutingProtocol>
46Ipv4DGRRoutingHelper::Create (Ptr<Node> node) const
47{
48 NS_LOG_LOGIC ("Adding DGRRouter interface to node " <<
49 node->GetId ());
50 // install DGRv2 Queue to netdevices
51
52 // install DGR router to node.
53 Ptr<DGRRouter> dgrRouter = CreateObject<DGRRouter> ();
54 node->AggregateObject (dgrRouter);
55
56 NS_LOG_LOGIC ("Adding DGRRouting Protocol to node " << node->GetId ());
57 Ptr<Ipv4DGRRouting> dgrRouting = CreateObject<Ipv4DGRRouting> ();
58 dgrRouter->SetRoutingProtocol (dgrRouting);

60 return dgrRouting;

100QueueDiscContainer
101Ipv4DGRRoutingHelper::Install (Ptr<NetDevice> d)
102{
103 QueueDiscContainer container;
104 // A TrafficControlLayer object is aggregated by the InternetStackHelper, but check
105 // anyway because a queue disc has no effect without a TrafficControlLayer object
106 Ptr<TrafficControlLayer> tc = d->GetNode ()->GetObject<TrafficControlLayer> ();
107 NS_ASSERT (tc);
108
109 // Generate the DGRv2Qeueu Object
110 ObjectFactory queueFactory;
111 queueFactory.SetTypeId ("ns3::DGRv2QueueDisc");
112 Ptr<DGRv2QueueDisc> qdisc = queueFactory.Create<DGRv2QueueDisc> ();
113 tc->SetRootQueueDiscOnDevice (d, qdisc);
114 container.Add (qdisc);
115 return container;
116}
